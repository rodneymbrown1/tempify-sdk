def merge_styles(*dicts: Dict[str, Any]) -> Dict[str, Any]:
    """
    Merge styles with precedence:
    global_defaults < layout_group < section.anchor.style < body.style
    """
    merged: Dict[str, Any] = {}
    for d in dicts:
        if not d:
            continue
        merged.update(d)
    return merged
